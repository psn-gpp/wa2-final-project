                 AND (?3 IS NULL OR m.defect = ?3)

                 AND (?5 IS NULL OR m.date = ?5) 
         """
     )
     fun findWithFilters(
         pageable: Pageable,
         vehicleId: Long,
         vehicleLicencePlate: String?,
         defect: String?,
         completedMaintenance: Boolean?,
         date: String?
     ): Page<MaintenanceHistory>
 }
