 package it.wa2.usermanagmentservice.dtos
 
 
 
 import it.wa2.usermanagmentservice.entities.GenericUser
 import jakarta.validation.constraints.Email
 import jakarta.validation.constraints.NotBlank
 import jakarta.validation.constraints.NotNull
 import jakarta.validation.constraints.Pattern
 
 
 data class GenericUserDTO(
     val id: Long,
 
     @field:NotBlank
     val name: String = "",
 
     @field:NotBlank
     val surname: String = "",
 
     @field:Email
     val email: String = "",
 
     @field:NotNull
     @field:NotBlank
     @field:Pattern(regexp = "^(\\+\\d{1,3}( )?)?((\\(\\d{3}\\))|\\d{3})[- .]?\\d{3}[- .]?\\d{4}$")
     val phone: String = "",
 
     @field:NotBlank
     val address: String = "",
 
     @field:NotBlank
     val city: String = "",
 
 )
 
 fun GenericUserDTO.toEntity() = GenericUser(
     name = this.name,
     surname = this.surname,
     email = this.email,
     phone = this.phone,
     address = this.address,
     city = this.city,
 )
